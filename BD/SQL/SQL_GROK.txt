-- Trigger para actualizar inventario en movimientos
CREATE TRIGGER TR_MovimientosInventario_Update
ON MovimientosInventario
AFTER INSERT
AS
BEGIN
    UPDATE inv
    SET inv.ExistenciaActual = inv.ExistenciaActual + 
        CASE WHEN i.TipoMovimiento IN ('Entrada', 'TraspasoEntrada') THEN i.Cantidad
             WHEN i.TipoMovimiento IN ('Salida', 'TraspasoSalida', 'AjusteNegativo') THEN -i.Cantidad
             ELSE 0 END,
        inv.UltimaActualizacion = GETDATE()
    FROM InventarioSucursal inv
    INNER JOIN inserted i ON inv.ProductoID = i.ProductoID AND inv.SucursalID = i.SucursalID;
END;
GO

-- Trigger para entrada de inventario al confirmar recepción
CREATE TRIGGER TR_RecepcionMercancia_Confirm
ON RecepcionMercancia
AFTER UPDATE
AS
BEGIN
    IF UPDATE(Confirmada)
    BEGIN
        DECLARE @RecepcionID INT;
        SELECT @RecepcionID = i.RecepcionID FROM inserted i WHERE i.Confirmada = 1;

        -- Actualizar inventario
        UPDATE inv
        SET inv.ExistenciaActual = inv.ExistenciaActual + d.Cantidad,
            inv.UltimaActualizacion = GETDATE()
        FROM InventarioSucursal inv
        INNER JOIN DetalleRecepcionMercancia d ON inv.ProductoID = d.ProductoID
        INNER JOIN RecepcionMercancia r ON d.RecepcionID = r.RecepcionID
        WHERE d.RecepcionID = @RecepcionID AND inv.SucursalID = r.SucursalID;

        -- Insertar movimiento
        INSERT INTO MovimientosInventario (ProductoID, SucursalID, TipoMovimiento, DocumentoReferencia, Cantidad, CostoUnitario, EmpleadoID)
        SELECT d.ProductoID, r.SucursalID, 'Entrada', r.NumeroRecepcion, d.Cantidad, d.CostoUnitario, r.EmpleadoID
        FROM DetalleRecepcionMercancia d
        INNER JOIN RecepcionMercancia r ON d.RecepcionID = r.RecepcionID
        WHERE d.RecepcionID = @RecepcionID;
    END
END;
GO

-- Trigger para transferencias (actualizado)
CREATE TRIGGER TR_ConfirmarTransferencia
ON Transferencias
AFTER UPDATE
AS
BEGIN
    IF UPDATE(Confirmada)
    BEGIN
        DECLARE @TransferenciaID INT, @SucOrigen INT, @SucDestino INT;
        SELECT @TransferenciaID = i.TransferenciaID, @SucOrigen = i.SucursalOrigenID, @SucDestino = i.SucursalDestinoID
        FROM inserted i WHERE i.Confirmada = 1;

        -- Restar de origen
        UPDATE invO
        SET invO.ExistenciaActual = invO.ExistenciaActual - dt.Cantidad,
            invO.UltimaActualizacion = GETDATE()
        FROM InventarioSucursal invO
        INNER JOIN DetalleTransferencias dt ON invO.ProductoID = dt.ProductoID
        WHERE dt.TransferenciaID = @TransferenciaID AND invO.SucursalID = @SucOrigen;

        -- Sumar a destino
        UPDATE invD
        SET invD.ExistenciaActual = invD.ExistenciaActual + dt.Cantidad,
            invD.UltimaActualizacion = GETDATE()
        FROM InventarioSucursal invD
        INNER JOIN DetalleTransferencias dt ON invD.ProductoID = dt.ProductoID
        WHERE dt.TransferenciaID = @TransferenciaID AND invD.SucursalID = @SucDestino;

        -- Movimientos
        INSERT INTO MovimientosInventario (ProductoID, SucursalID, TipoMovimiento, DocumentoReferencia, Cantidad, CostoUnitario, EmpleadoID)
        SELECT dt.ProductoID, @SucOrigen, 'TraspasoSalida', t.NumeroTransferencia, -dt.Cantidad, dt.CostoUnitario, t.EmpleadoID
        FROM DetalleTransferencias dt
        INNER JOIN Transferencias t ON dt.TransferenciaID = t.TransferenciaID
        WHERE dt.TransferenciaID = @TransferenciaID;

        INSERT INTO MovimientosInventario (ProductoID, SucursalID, TipoMovimiento, DocumentoReferencia, Cantidad, CostoUnitario, EmpleadoID)
        SELECT dt.ProductoID, @SucDestino, 'TraspasoEntrada', t.NumeroTransferencia, dt.Cantidad, dt.CostoUnitario, t.EmpleadoID
        FROM DetalleTransferencias dt
        INNER JOIN Transferencias t ON dt.TransferenciaID = t.TransferenciaID
        WHERE dt.TransferenciaID = @TransferenciaID;
    END
END;
GO

ON DetalleFactura
AFTER INSERT
AS
BEGIN
    DECLARE @SucursalID INT;
    SELECT @SucursalID = f.SucursalID FROM Facturas f INNER JOIN inserted i ON f.FacturaID = i.FacturaID;

    UPDATE inv
    SET inv.ExistenciaActual = inv.ExistenciaActual - i.Cantidad,
        inv.UltimaActualizacion = GETDATE()
    FROM InventarioSucursal inv
    INNER JOIN inserted i ON inv.ProductoID = i.ProductoID
    WHERE inv.SucursalID = @SucursalID;

    INSERT INTO MovimientosInventario (ProductoID, SucursalID, TipoMovimiento, DocumentoReferencia, Cantidad, CostoUnitario)
    SELECT i.ProductoID, @SucursalID, 'Salida', f.NumeroFactura, -i.Cantidad, p.CostoUnitario
    FROM inserted i
    INNER JOIN Facturas f ON i.FacturaID = f.FacturaID
    INNER JOIN Productos p ON i.ProductoID = p.ProductoID;
END;
GO

-- Trigger para devoluciones (entrada/salida inventario)
CREATE TRIGGER TR_Devoluciones_Insert
ON DetalleDevoluciones
AFTER INSERT
AS
BEGIN
    DECLARE @SucursalID INT, @Tipo VARCHAR(20);
    SELECT @SucursalID = f.SucursalID, @Tipo = d.TipoDevolucion
    FROM Devoluciones d
    INNER JOIN inserted i ON d.DevolucionID = i.DevolucionID
    INNER JOIN Facturas f ON d.FacturaID = f.FacturaID;

    IF @Tipo = 'ClienteOptica'
    BEGIN
        -- Entrada a inventario
        UPDATE inv
        SET inv.ExistenciaActual = inv.ExistenciaActual + i.Cantidad
        FROM InventarioSucursal inv
        INNER JOIN inserted i ON inv.ProductoID = i.ProductoID
        WHERE inv.SucursalID = @SucursalID;
    END
    ELSE IF @Tipo = 'OpticaProveedor'
    BEGIN
        -- Salida de inventario
        UPDATE inv
        SET inv.ExistenciaActual = inv.ExistenciaActual - i.Cantidad
        FROM InventarioSucursal inv
        INNER JOIN inserted i ON inv.ProductoID = i.ProductoID
        WHERE inv.SucursalID = @SucursalID;
    END
END;
GO

-- VISTAS PARA REPORTES ESTADÍSTICOS Y AUDITORÍAS
-- =============================================
CREATE VIEW VW_VentasGenerales AS
SELECT f.FacturaID, f.NumeroFactura, c.Nombre + ' ' + c.Apellido AS Cliente, e.Nombre + ' ' + e.Apellido AS Vendedor,
       f.FechaFactura, f.Total, f.Status
FROM Facturas f
INNER JOIN Clientes c ON f.ClienteID = c.ClienteID
INNER JOIN Empleados e ON f.EmpleadoID = e.EmpleadoID;
GO

CREATE VIEW VW_InventarioActual AS
SELECT s.Nombre AS Sucursal, p.Descripcion AS Producto, i.ExistenciaActual, i.ExistenciaMinima
FROM InventarioSucursal i
INNER JOIN Sucursales s ON i.SucursalID = s.SucursalID
INNER JOIN Productos p ON i.ProductoID = p.ProductoID;
GO

CREATE VIEW VW_CuentasPorCobrarDetallada AS
SELECT cc.CuentaCobrarID, c.Nombre + ' ' + c.Apellido AS Cliente, cc.MontoTotal, cc.Saldo, cc.FechaVencimiento
FROM CuentasPorCobrar cc
INNER JOIN Clientes c ON cc.ClienteID = c.ClienteID;
GO

CREATE VIEW VW_AuditoriaVentas AS
SELECT a.AuditoriaID, u.NombreUsuario, a.Modulo, a.Accion, a.FechaHora
FROM Auditoria a
INNER JOIN Usuarios u ON a.UsuarioID = u.UsuarioID
WHERE a.Modulo = 'Ventas';
GO
